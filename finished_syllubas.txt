\section{Course Description}
\subsubsection{This course is a study of the fundamentals of baking including, dough, quick breads, pies, cakes, cookies, tarts and basic items made in a bakery.  Topics include baking terminology, tool and equipment use, formula conversions, functions of ingredients, and the use of proper flours. }
\subsubsection{Time: F14:00 - 17:00}
\subsubsection{Place: Evans 330}
\subsubsection{Instructor: Ophelia Wang}
\subsubsection{Instructor's Email: wangyumeng2017@berkeley.edu}

\section{Course Objectives}
\subsection{Knowledge Based}
\subsubsection{Identify and explain baking terms, ingredients, equipment and tools.}
\subsubsection{Employ safe food handling practices using contemporary guidelines}
\subsection{Skills Based}
\subsubsection{Scale and measure ingredients.}
\subsubsection{Prepare yeast dough, quick breads, pies, cookies, cakes, icing, pate choux, and savory baking.}
\subsubsection{Produce baked products using commercial ingredients and equipment.}
